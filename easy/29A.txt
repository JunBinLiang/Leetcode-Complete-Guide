#include <iostream>
#include <vector>
using namespace std;



int main()
{
    vector<vector<int>>A;
    int n;
    cin>>n;
    while(n--){
        int a,b;
        cin>>a>>b;
        A.push_back({a,b});
    }
    
    for(int i=0;i<A.size();i++){
        int a=A[i][0]+A[i][1];
        for(int j=i+1;j<A.size();j++){
            int b=A[j][0]+A[j][1];
            if(b==A[i][0]&&a==A[j][0]){
                cout<<"YES"<<endl;return 0;
            }
        }
    }
    cout<<"NO";
    return 0;
}
